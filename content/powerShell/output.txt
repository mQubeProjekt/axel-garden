$FormsAssembly = [System.Reflection.Assembly]::LoadWithPartialName("System.Windows.Forms")

function New-WinFormObject {
    param([string]$TypeName)

    $type = $FormsAssembly.GetType("System.Windows.Forms.$TypeName")
    return [Activator]::CreateInstance($type)
}

$form = New-WinFormObject "Form"
$form.Text = "Test"
$form.Width = 400
$form.Height = 200
$form.ShowDialog()
